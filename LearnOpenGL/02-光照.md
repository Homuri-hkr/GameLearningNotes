### 基础光照

**Binn-Phong基于世界空间着色**

```glsl
//shader.vs
#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aNormal;

out vec3 Normal;
out vec3 FragPos;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

void main(){
	gl_Position = projection * view * model * vec4(aPos, 1.0f);
	//计算逆矩阵是个开销大的运算实际应放在CPU上计算
	Normal = mat3(transpose(inverse(model))) * aNormal;
	FragPos = vec3(model * vec4(aPos, 1.0f));
}
```

```glsl
//shader.fs
#version 330 core
in vec3 Normal;
in vec3 FragPos;

out vec4 FragColor;

uniform vec3 objectColor;
uniform vec3 lightColor;

uniform vec3 lightPos;
uniform vec3 viewPos;

void main(){
	//Binn-Phong反射模型：L = k_a*I_a + k_d(I/r^2)max(0,nl)+k_s(I/r^2)max(0,nh)^p
	//这里是简化的版本，没有除以距离的平方
	//环境光
	float ambientStrength = 0.1f;
	vec3 ambient = ambientStrength * lightColor;
	//漫反射
	vec3 norm = normalize(Normal);
	vec3 lightDir = normalize(lightPos - FragPos);
	float diff = max(dot(norm, lightDir), 0.0f);		//使用max保证diff不为负数
	vec3 diffuse = diff * lightColor;
	//镜面反射
	float specularStength = 0.5f;
	vec3 viewDir = normalize(viewPos - FragPos);
	//reflect要求冲光源指向片段的向量
	vec3 reflectDir = reflect(-lightDir, norm);
	//32是高光的反光度，越高反射光中间区域亮度越强越集中
	float spec = pow(max(dot(viewDir, reflectDir), 0.0f), 256);
	vec3 specular = specularStength * spec * lightColor;
	
	vec3 phong = ambient + diffuse + specular;
	FragColor = vec4(phong * objectColor, 1.0f);
}
```

```glsl
in vec3 Normal;
in vec3 FragPos;
in vec3 LightPos;

out vec4 FragColor;

uniform vec3 objectColor;
uniform vec3 lightColor;
uniform vec3 viewPos;

void main(){
	//Binn-Phong反射模型：L = k_a*I_a + k_d(I/r^2)max(0,nl)+k_s(I/r^2)max(0,nh)^p
	//这里是简化的版本，没有除以距离的平方
	//环境光

	vec3 ambient = ambientStrength * lightColor;

	vec3 norm = normalize(Normal);
	vec3 lightDir = normalize(LightPos - FragPos);
	float diff = max(dot(norm, lightDir), 0.0f);		//使用max保证diff不为负数
	vec3 diffuse = diff * lightColor;
	//镜面反射
	float specularStength = 0.5f;
	vec3 viewDir = normalize(viewPos - FragPos);
	//reflect要求冲光源指向片段的向量
	vec3 reflectDir = reflect(-lightDir, norm);
	//32是高光的反光度，越高反射光中间区域亮度越强越集中
	float spec = pow(max(dot(viewDir, reflectDir), 0.0f), 256);
	vec3 specular = specularStength * spec * lightColor;
	
	vec3 phong = ambient + diffuse + specular;
	FragColor = vec4(phong * objectColor, 1.0f);
```

```glgl
//shader.fs
#version 330 core
```
```glsl
struct Material{
```
```glsl
	//环境光颜色几乎所有情况下都等于漫反射颜色
	sampler2D diffuse;
	sampler2D specular;
	float shininess;
};
uniform Material material;

struct Light{
	vec3 position;
	vec3 ambient;
	vec3 diffuse;
	vec3 specular;
};
uniform Light light;

in vec3 Normal;
in vec3 FragPos;
in vec2 TexCoords;

out vec4 FragColor;

uniform vec3 viewPos;

void main(){
    vec3 ambient = light.ambient * texture(material.diffuse, TexCoords).rgb;
	
	vec3 norm = normalize(Normal);
	vec3 lightDir = normalize(light.position - FragPos);
	float diff = max(0.0f, dot(norm, lightDir));
	vec3 diffuse = light.diffuse * diff * texture(material.diffuse, TexCoords).rgb;
	
	vec3 viewDir = normalize(viewPos - FragPos);
	vec3 reflectDir = reflect(-lightDir, norm);
	float spec = pow(max(0.0f, dot(viewDir, reflectDir)), material.shininess);
```
```glsl
	vec3 specular = light.specular * spec * (vec3(texture(material.specular, TexCoords)));
	
	vec3 Phong = ambient + diffuse + specular;
	FragColor = vec4(Phong, 1.0f);
}
```

```glsl
//shader.fs
#version 330 core
struct Material{
	//环境光颜色几乎所有情况下都等于漫反射颜色
	sampler2D diffuse;
	sampler2D specular;
	float shininess;
};
uniform Material material;

struct Light{
	//点光源
	vec3 position;
	//通常为1.0，为了保证分母不小于1
	float constant;
	//一次项，线性减少强度
	float linear;
	//二次项，在近距离快速减少强度（因为其小于1
	float quadratic;
	
	vec3 ambient;
	vec3 diffuse;
	vec3 specular;
};
uniform Light light;

in vec3 Normal;
in vec3 FragPos;
in vec2 TexCoords;

out vec4 FragColor;

uniform vec3 viewPos;

void main(){
	float distance = length(light.position - FragPos);
	float attenuation = 1.0f / (light.constant + light.linear * distance + light.quadratic * (distance * distance));

    vec3 ambient = light.ambient * texture(material.diffuse, TexCoords).rgb;
	
	vec3 norm = normalize(Normal);
	vec3 lightDir = normalize(light.position - FragPos);
	float diff = max(0.0f, dot(norm, lightDir));
	vec3 diffuse = light.diffuse * diff * texture(material.diffuse, TexCoords).rgb;
	
	vec3 viewDir = normalize(viewPos - FragPos);
	vec3 reflectDir = reflect(-lightDir, norm);
	float spec = pow(max(0.0f, dot(viewDir, reflectDir)), material.shininess);
	vec3 specular = light.specular * spec * (vec3(texture(material.specular, TexCoords)));
	
	//环境光也要保证衰减，因为若多余一个光源，场景的环境光就会叠加
	vec3 Phong = (ambient + diffuse + specular) * attenuation;
	FragColor = vec4(Phong, 1.0f);
}
```

```glsl
//shader.fs
#version 330 core
struct Material{
	//环境光颜色几乎所有情况下都等于漫反射颜色
	sampler2D diffuse;
	sampler2D specular;
	float shininess;
};
uniform Material material;

struct Light{
	//聚光灯
	vec3 position;
	vec3 direction;
	float cutOff;
	float outerCutOff;
	
	float constant;
	float linear;
	float quadratic;
	
	vec3 ambient;
	vec3 diffuse;
	vec3 specular;
};
uniform Light light;

in vec3 Normal;
in vec3 FragPos;
in vec2 TexCoords;

out vec4 FragColor;

uniform vec3 viewPos;

void main(){
    vec3 ambient = light.ambient * texture(material.diffuse, TexCoords).rgb;
	
	vec3 norm = normalize(Normal);
	vec3 lightDir = normalize(light.position - FragPos);
	float diff = max(0.0f, dot(norm, lightDir));
	vec3 diffuse = light.diffuse * diff * texture(material.diffuse, TexCoords).rgb;
	
	vec3 viewDir = normalize(viewPos - FragPos);
	vec3 reflectDir = reflect(-lightDir, norm);
	float spec = pow(max(0.0f, dot(viewDir, reflectDir)), material.shininess);
	vec3 specular = light.specular * spec * (vec3(texture(material.specular, TexCoords)));
	//计算片段和光的夹角
	float theta = dot(lightDir, normalize(-light.direction));
	float epsilon = light.cutOff - light.outerCutOff;
	float intensity = clamp((theta - light.outerCutOff) / epsilon, 0.0f, 1.0f);
		
	float distance = length(light.position - FragPos);
	float attenuation = 1.0f / (light.constant + light.linear * distance + light.quadratic * (distance * distance));
	//不对环境光做影响，保证没被照射到的地方还有环境光	
	vec3 Phong = (ambient + (diffuse + specular) * intensity) * attenuation;
	FragColor = vec4(Phong, 1.0f);
}
```

```glsl
	float diff = max(dot(lightDir, normal), 0.0f);
	
	vec3 reflectDir = reflect(-lightDir, normal);
	float spec = pow(max(0.0f, dot(reflectDir, viewDir)), material.shininess);
	
	vec3 ambient = light.ambient * vec3(texture(material.diffuse, TexCoords));
	vec3 diffuse = light.diffuse * diff * vec3(texture(material.diffuse, TexCoords));
	vec3 specular = light.specular * spec * vec3(texture(material.specular, TexCoords));
	return ambient + diffuse + specular;
}
//计算点光源函数
vec3 CalcPointLight(PointLight light, vec3 normal, vec3 fragPos, vec3 viewDir){
	vec3 lightDir = normalize(light.position - fragPos);
	
	float diff = max(0.0f, dot(lightDir, normal));
	
	vec3 reflectDir = reflect(-lightDir, normal);
	float spec = pow(max(0.0f, dot(viewDir, reflectDir)), material.shininess);
	
	float distance = length(light.position - fragPos);
	float attenuation = 1.0 / (light.constant + light.linear * distance + light.quadratic * (distance * distance));
	
	vec3 ambient = light.ambient * vec3(texture(material.diffuse, TexCoords));
	vec3 diffuse = light.diffuse * diff * vec3(texture(material.diffuse, TexCoords));
	vec3 specular = light.specular * spec * vec3(texture(material.specular, TexCoords));
	return (ambient + diffuse + specular) * attenuation;
}
//计算聚光源函数
vec3 CalcSpotLight(SpotLight light, vec3 normal, vec3 fragPos, vec3 viewDir){
	vec3 norm = normalize(normal);
	vec3 lightDir = normalize(light.position - FragPos);
	float diff = max(0.0f, dot(norm, lightDir));
	
	vec3 reflectDir = reflect(-lightDir, norm);
	float spec = pow(max(0.0f, dot(reflectDir, viewDir)), material.shininess);
	
	float theta = dot(lightDir, normalize(-light.direction));
	float epsilon = light.cutOff - light.outerCutOff;
	float intensity = clamp((theta - light.outerCutOff) / epsilon, 0.0f, 1.0f);
	
	float distance = length(light.position - fragPos);
	float attenuation = 1.0f / (light.constant + light.linear * distance + light.quadratic *(distance * distance));
	
	vec3 ambient = light.ambient * vec3(texture(material.diffuse, TexCoords));;
	vec3 diffuse = light.diffuse * diff * vec3(texture(material.diffuse, TexCoords));
	vec3 specular = light.specular * spec * vec3(texture(material.specular, TexCoords));
	
	return (ambient + (diffuse + specular) * intensity) * attenuation;
}
void main(){
    vec3 norm = normalize(Normal);
	vec3 viewDir = normalize(viewPos - FragPos);
	//平行光
	vec3 result = CalcDirLight(dirLight, norm, viewDir);
	//点光
	for(int i = 0; i < NR_POINT_LIGHTS; i++){
		result += CalcPointLight(pointLights[i], norm, FragPos, viewDir);
	}
	//聚光
	result += CalcSpotLight(spotLight, norm, FragPos, viewDir);
	FragColor = vec4(result, 1.0f);
}
```
